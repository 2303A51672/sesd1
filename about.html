JavaScript
/**
 * Debounced logger — waits until user stops calling `log` for 300ms:
 */
function debounce(fn, ms = 300){
  let t = null;
  return (...args) => {
    clearTimeout(t);
    t = setTimeout(()=> fn(...args), ms);
  };
}

const log = debounce((msg)=> console.log('LOG:', msg), 300);

log('first');
log('still typing...');
setTimeout(()=> log('done'), 500);
Python
def fibonacci(n):
    """Return first n Fibonacci numbers as a list."""
    if n <= 0:
        return []
    seq = [0, 1]
    while len(seq) < n:
        seq.append(seq[-1] + seq[-2])
    return seq[:n]

if __name__ == '__main__':
    print(fibonacci(10))
